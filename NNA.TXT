/**
 * Automatically updates the onboarding dropdown checklist.
 * Reads all entries from the Enumeration tab and subtracts those in Onboarding.
 */
function syncAndFilterOutlets() {
  var ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  
  // 1. Fetch all enumerated business names (Column B of Enumeration Sheet)
  var enumSheet = ss.getSheetByName(ENUMERATION_SHEET_NAME);
  if (!enumSheet) {
    Logger.log("Error: Enumeration sheet tab not found. Make sure you renamed the tab.");
    return;
  }
  var enumLastRow = enumSheet.getLastRow();
  var enumeratedOutlets = [];
  
  if (enumLastRow > 1) {
    var enumValues = enumSheet.getRange(2, 2, enumLastRow - 1, 1).getValues();
    for (var i = 0; i < enumValues.length; i++) {
      var name = enumValues[i][0].toString().trim();
      if (name && enumeratedOutlets.indexOf(name) === -1) {
        enumeratedOutlets.push(name);
      }
    }
  }

  // 2. Fetch all already onboarded business names (Column B of Onboarding Sheet)
  var onboardSheet = ss.getSheetByName(ONBOARDING_SHEET_NAME);
  var onboardedOutlets = [];
  if (onboardSheet) {
    var onboardLastRow = onboardSheet.getLastRow();
    if (onboardLastRow > 1) {
      var onboardValues = onboardSheet.getRange(2, 2, onboardLastRow - 1, 1).getValues();
      for (var j = 0; j < onboardValues.length; j++) {
        var oName = onboardValues[j][0].toString().trim();
        if (oName) {
          onboardedOutlets.push(oName);
        }
      }
    }
  }

  // 3. Filter Outlets: Only keep items that are Enumerated BUT NOT yet Onboarded
  var availableOutlets = enumeratedOutlets.filter(function(outlet) {
    return onboardedOutlets.indexOf(outlet) === -1;
  });

  if (availableOutlets.length === 0) {
    availableOutlets = ["No pending outlets awaiting onboarding"];
  }

  // 4. Update the "Verified Outlet Name" question in the Onboarding Form
  var form = FormApp.openById(ONBOARDING_FORM_ID);
  var items = form.getItems();
  var updated = false;

  for (var k = 0; k < items.length; k++) {
    if (items[k].getTitle() === "Verified Outlet Name" && items[k].getType() === FormApp.ItemType.LIST) {
      var listItem = items[k].asListItem();
      listItem.setChoiceValues(availableOutlets);
      updated = true;
      Logger.log("Successfully updated Onboarding Dropdown. Available options count: " + availableOutlets.length);
      break;
    }
  }
  
  if (!updated) {
    Logger.log("Warning: Could not find a Dropdown list item titled 'Verified Outlet Name' to update.");
  }
}

/**
 * Helper function to cleanly change the raw text input to a dynamic dropdown selector.
 */
function convertTextBoxToDropdown() {
  var form = FormApp.openById(ONBOARDING_FORM_ID);
  var items = form.getItems();
  
  for (var i = 0; i < items.length; i++) {
    if (items[i].getTitle() === "Verified Outlet Name") {
      if (items[i].getType() === FormApp.ItemType.TEXT) {
        var index = items[i].getIndex();
        form.deleteItem(items[i]);
        var dropdown = form.addListItem().setTitle("Verified Outlet Name")
                           .setHelpText("Select from the list of verified field entries. Completed items are dynamically removed.")
                           .setRequired(true);
        form.moveItem(dropdown.getIndex(), index);
        Logger.log("Converted text input field to dynamic list selector successfully.");
      }
      break;
    }
  }
  syncAndFilterOutlets();
}

  configureEnumerationForm(ENUMERATION_FORM_ID);

}

function clearFormItems(formId) {
  var form = FormApp.openById(formId);
  var items = form.getItems();
  for (var i = items.length - 1; i >= 0; i--) {
    form.deleteItem(items[i]);
  }
  Logger.log('Successfully cleared form ID: ' + formId);
}
